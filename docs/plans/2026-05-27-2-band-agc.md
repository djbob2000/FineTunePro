# Optimod-Style 2-Band AGC Implementation Plan

> **For Antigravity:** REQUIRED WORKFLOW: Use `.agent/workflows/execute-plan.md` to execute this plan in single-flow mode.

**Goal:** Upgrade the Loudness Equalizer (AGC) to operate in a 2-band mode (Bass and Master) split at 150 Hz using a 4th-order Linkwitz-Riley crossover, with Bass-to-Master band coupling set to -3 dB.

**Architecture:** 
1. Re-introduce the 4th-order Linkwitz-Riley crossover filters to split audio into Bass (0–150 Hz) and Master (150–Nyquist Hz) bands.
2. Structure the AGC to run two independent band processors (Bass and Master) for gain detection and envelope tracking, driven by a global broadband silence gate.
3. Apply a -3 dB Bass-to-Master coupling clamp (`gains[bass] = min(gains[bass], gains[master] + 3.0)`) before applying gains to the band-limited signals and summing them back together.

**Tech Stack:** Swift, Core Audio, Accelerate (vDSP)

```swift
    /// Compute low-pass biquad coefficients (RBJ cookbook, 2nd-order Butterworth)
    /// Returns [b0, b1, b2, a1, a2] normalized by a0 for vDSP_biquad
    static func lowPassCoefficients(
        frequency: Double,
        q: Double,
        sampleRate: Double
    ) -> [Double] {
        let omega = 2.0 * .pi * frequency / sampleRate
        let sinW = sin(omega)
        let cosW = cos(omega)
        let alpha = sinW / (2.0 * q)

        let b0 =  (1.0 - cosW) / 2.0
        let b1 =  1.0 - cosW
        let b2 =  (1.0 - cosW) / 2.0
        let a0 =  1.0 + alpha
        let a1 = -2.0 * cosW
        let a2 =  1.0 - alpha

        return [b0 / a0, b1 / a0, b2 / a0, a1 / a0, a2 / a0]
    }
```

```swift
import Foundation

struct BiquadSection {
    private var b0: Float, b1: Float, b2: Float
    private var a1: Float, a2: Float
    private var z1: Float = 0, z2: Float = 0

    init(coefficients: [Double]) {
        precondition(coefficients.count == 5)
        b0 = Float(coefficients[0])
        b1 = Float(coefficients[1])
        b2 = Float(coefficients[2])
        a1 = Float(coefficients[3])
        a2 = Float(coefficients[4])
    }

    @inline(__always)
    mutating func process(_ sample: Float) -> Float {
        let y = b0 * sample + z1
        z1 = b1 * sample - a1 * y + z2
        z2 = b2 * sample - a2 * y
        return y
    }

    mutating func reset() {
        z1 = 0; z2 = 0
    }
}

struct LR4Filter {
    private var stage1: BiquadSection
    private var stage2: BiquadSection

    init(lowPass frequency: Double, sampleRate: Double) {
        let q = 1.0 / sqrt(2.0)
        let coeffs1 = BiquadMath.lowPassCoefficients(frequency: frequency, q: q, sampleRate: sampleRate)
        let coeffs2 = BiquadMath.lowPassCoefficients(frequency: frequency, q: q, sampleRate: sampleRate)
        stage1 = BiquadSection(coefficients: coeffs1)
        stage2 = BiquadSection(coefficients: coeffs2)
    }

    init(highPass frequency: Double, sampleRate: Double) {
        let q = 1.0 / sqrt(2.0)
        let coeffs1 = BiquadMath.highPassCoefficients(frequency: frequency, q: q, sampleRate: sampleRate)
        let coeffs2 = BiquadMath.highPassCoefficients(frequency: frequency, q: q, sampleRate: sampleRate)
        stage1 = BiquadSection(coefficients: coeffs1)
        stage2 = BiquadSection(coefficients: coeffs2)
    }

    @inline(__always)
    mutating func process(_ sample: Float) -> Float {
        return stage2.process(stage1.process(sample))
    }

    mutating func reset() {
        stage1.reset(); stage2.reset()
    }
}

struct LinkwitzRileyCrossover2 {
    private var lp4: LR4Filter
    private var hp4: LR4Filter

    init(frequency: Double, sampleRate: Double) {
        lp4 = LR4Filter(lowPass: frequency, sampleRate: sampleRate)
        hp4 = LR4Filter(highPass: frequency, sampleRate: sampleRate)
    }

    @inline(__always)
    mutating func process(_ sample: Float) -> (Float, Float) {
        let low = lp4.process(sample)
        let high = hp4.process(sample)
        return (low, high)
    }

    mutating func reset() {
        lp4.reset()
        hp4.reset()
    }
}
```

```swift
    @inline(__always)
    private static func applySjp(
        jumpProtection: Bool,
        drivenPeak: Float,
        targetDb: Float,
        sjpAttackCoeff: Float,
        currentGainDb: Float
    ) -> Float {
        var gainDb = currentGainDb
        if jumpProtection && drivenPeak > 0 {
            let drivenPeakDb = LoudnessEqualizerMath.linearToDb(drivenPeak)
            let outputDb = drivenPeakDb + gainDb
            let jumpThresholdDb = targetDb + 4.0
            if outputDb > jumpThresholdDb {
                let targetSjpGainDb = jumpThresholdDb - drivenPeakDb
                gainDb += sjpAttackCoeff * (targetSjpGainDb - gainDb)
            }
        }
        return gainDb
    }
```

5. Modify `processAgcSample` in `LoudnessEqualizer.swift` to process the state of a single band, accepting a `broadbandGatingFactor` to link the silence gate:
```swift
    @inline(__always)
    private func processBandSample(
        band: inout AgcBandState,
        weighted: Float,
        drivenPeak: Float,
        targetDb: Float,
        attackNorm: Float,
        releaseNorm: Float,
        jumpProtection: Bool,
        broadbandGatingFactor: Float,
        silenceGateSlowdown: Float,
        gateSlowdownFactor: Float,
        suddenDropProtection: Bool,
        suddenDropThresholdDb: Float,
        suddenDropSpeedup: Float,
        agcWindowSize: Float,
        progressiveRatioEnabled: Bool,
        minRatio: Float,
        maxRatio: Float,
        progressiveRate: Float,
        targetIdleGain: Float
    ) {
        // 1. Envelope detection
        let sampleSq = weighted * weighted
        if sampleSq > band.envelopeSq {
            band.envelopeSq += envAttackCoeff * (sampleSq - band.envelopeSq)
        } else {
            band.envelopeSq += envReleaseCoeff * (sampleSq - band.envelopeSq)
        }
        let levelDb = LoudnessEqualizerMath.meanSquareToDb(band.envelopeSq)

        // 2. AGC Decision with broadband gating
        let delta = levelDb - targetDb
        let halfWindow = agcWindowSize * 0.5
        let gatingFactor = broadbandGatingFactor

        if delta > halfWindow {
            // signal above window
            let overshootDb = delta - halfWindow
            var targetGainDb = -delta
            
            if progressiveRatioEnabled && minRatio > 1.0 {
                let sMin = 1.0 / minRatio
                let sMax = maxRatio > 0.0 ? (1.0 / maxRatio) : 0.0
                let s = sMax + (sMin - sMax) * exp(-progressiveRate * max(overshootDb, 0.0))
                targetGainDb = -halfWindow - overshootDb * (1.0 - s)
            }

            if targetGainDb < band.currentGainDb {
                // attack
                band.currentGainDb -= attackNorm * max(overshootDb, 0)
                if band.currentGainDb < targetGainDb { band.currentGainDb = targetGainDb }
            } else if targetGainDb > band.currentGainDb {
                // recovery
                var speedMult: Float = 1.0
                if levelDb < silenceGateSlowdown {
                    speedMult *= pow(gateSlowdownFactor, 1.0 - gatingFactor)
                }
                let activeAlpha = releaseNorm * speedMult
                let effectiveTargetGainDb = gatingFactor * targetGainDb + (1.0 - gatingFactor) * targetIdleGain
                let effectiveAlpha = gatingFactor * activeAlpha + (1.0 - gatingFactor) * (1.0 - gatingFactor) * fallbackAlpha
                band.currentGainDb += (effectiveTargetGainDb - band.currentGainDb) * effectiveAlpha
                if band.currentGainDb > targetGainDb { band.currentGainDb = targetGainDb }
                if band.currentGainDb > 0 { band.currentGainDb = 0 }
            }
        } else if delta < -halfWindow {
            // signal below window (release)
            let activeTargetDb: Float = 0.0
            var speedMult: Float = 1.0
            if levelDb < silenceGateSlowdown {
                speedMult *= pow(gateSlowdownFactor, 1.0 - gatingFactor)
            }
            if suddenDropProtection && delta < -suddenDropThresholdDb {
                speedMult *= suddenDropSpeedup
            }
            let activeAlpha = releaseNorm * speedMult
            let effectiveTargetGainDb = gatingFactor * activeTargetDb + (1.0 - gatingFactor) * targetIdleGain
            let effectiveAlpha = gatingFactor * activeAlpha + (1.0 - gatingFactor) * (1.0 - gatingFactor) * fallbackAlpha
            band.currentGainDb += (effectiveTargetGainDb - band.currentGainDb) * effectiveAlpha
            if band.currentGainDb > 0 { band.currentGainDb = 0 }
        } else {
            // inside window
            if fallbackAlpha > 0 {
                let effectiveAlpha = (1.0 - gatingFactor) * (1.0 - gatingFactor) * fallbackAlpha
                band.currentGainDb += (targetIdleGain - band.currentGainDb) * effectiveAlpha
                if band.currentGainDb > 0 { band.currentGainDb = 0 }
            }
        }

        // 3. SJP
        band.currentGainDb = Self.applySjp(
            jumpProtection: jumpProtection,
            drivenPeak: drivenPeak,
            targetDb: targetDb,
            sjpAttackCoeff: sjpAttackCoeff,
            currentGainDb: band.currentGainDb
        )
        band.currentGainLinear = LoudnessEqualizerMath.dbToLinear(band.currentGainDb)
    }
```
